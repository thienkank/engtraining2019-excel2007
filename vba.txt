Sub insertARowAbove(control As IRibbonControl)
    ActiveCell.EntireRow.Select
    On Error GoTo Last
    Selection.Insert Shift:=xlToDown, CopyOrigin:=xlFormatFromRightorAbove
Last:     Exit Sub
End Sub
Sub insertAColumnLeft(control As IRibbonControl)
    ActiveCell.EntireColumn.Select
    On Error GoTo Last
    Selection.Insert Shift:=xlToRight, CopyOrigin:=xlFormatFromRightorAbove
Last:     Exit Sub
End Sub
Sub insertRowsWithOneRow(control As IRibbonControl)
    Dim answer As Variant
LableNumber:
    answer = Application.InputBox("กรอกจำนวนแถว", "เพิ่มข้อมูลโดยคัดลอกจากแถวเดียว", , , , , , 1)
    If IsNumeric(answer) Then
        If IsEmpty(answer) Then
            MsgBox "ข้อมูลที่กรอกควรมีค่าตั้งแต่ 1 ขึ้นไป", vbOKOnly, "เพิ่มข้อมูลโดยคัดลอกจากแถวเดียวที่เลือก"
            GoTo LableNumber:
        Else
            If answer < 0 Then
                MsgBox "ข้อมูลที่กรอกควรมีค่าตั้งแต่ 1 ขึ้นไป", vbOKOnly, "เพิ่มข้อมูลโดยคัดลอกจากแถวเดียวที่เลือก"
                GoTo LableNumber:
            ElseIf answer > 0 Then
                ActiveCell.EntireRow.Copy
                Range(ActiveCell.Offset(1, 0), ActiveCell.Offset(answer, 0)).EntireRow.Insert Shift:=xlDown
                Application.CutCopyMode = False
            End If
        End If
    End If
End Sub
Sub AutoFitRow(control As IRibbonControl)
    Dim selectedCell As Range
    Set selectedCell = Application.ActiveCell
    Range(Cells(selectedCell.Row, selectedCell.Column), Cells(selectedCell.Row, selectedCell.Column)).Select
    Selection.Rows.AutoFit
End Sub
Sub AutoFitAllRows(control As IRibbonControl)
    Rows().AutoFit
End Sub
Sub AutoFitColumn(control As IRibbonControl)
    Dim selectedCell As Range
    Set selectedCell = Application.ActiveCell
    Range(Cells(selectedCell.Row, selectedCell.Column), Cells(selectedCell.Row, selectedCell.Column)).Select
    Selection.Columns.AutoFit
End Sub
Sub AutoFitAllColumns(control As IRibbonControl)
    Columns().AutoFit
End Sub
